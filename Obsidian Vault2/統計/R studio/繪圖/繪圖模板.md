### 🏆 R 語言自動化繪圖．終極通用模板 (Universal Plotting Template)

請將這份程式碼儲存起來。未來無論是畫 Bar Chart 還是 Line Chart，你只需要修改最上面的「開關設定」即可。

#### 1. 環境設定與通用主題 (The Foundation)

這部分負責定義「長相」與「工具」。

R

```
library(ggplot2)
library(tidyverse)
library(scales) # 關鍵自動化套件

# ==============================================================================
# ==== A. 通用外觀設定 (Theme & Colors) ====
# ==============================================================================

# 1. 定義顏色 (A/B 組)
my_colors <- c("A" = "#31688E", "B" = "#E67E22")
my_shapes <- c("A" = 16, "B" = 17)

# 2. 定義統一主題 (解決字體大小、背景、軸線問題)
theme_scientific <- function() {
  theme_classic(base_size = 14) + # 基礎字體 14
    theme(
      # 標題設定
      plot.title = element_text(face = "bold", size = 18, hjust = 0.5),
      plot.subtitle = element_text(size = 12, color = "gray40"),
      
      # 座標軸設定 (黑色實線, 加粗)
      axis.line = element_line(color = "black", linewidth = 1),
      axis.ticks = element_line(color = "black", linewidth = 1),
      axis.text = element_text(color = "black", size = 14, face = "bold"),
      axis.title = element_text(face = "bold", size = 16),
      
      # 圖例設定 (內嵌右上角)
      legend.position = c(0.95, 0.98),
      legend.justification = c("right", "top"),
      legend.background = element_rect(fill = "white", color = "black", linewidth = 0.2),
      legend.title = element_blank(),
      legend.text = element_text(size = 12)
    )
}

# 3. 顯著性標記位置計算函數 (解決星星高度問題)
# 邏輯：最高點 + 全距的 5% = 星星底部
calc_star_y <- function(df_sub) {
  raw_max <- max(df_sub$avg + df_sub$se, na.rm = TRUE)
  raw_min <- min(df_sub$avg - df_sub$se, na.rm = TRUE)
  gap <- (raw_max - raw_min) * 0.05
  if(gap == 0) gap <- raw_max * 0.05 # 防呆
  return(gap)
}
```

#### 2. 自動化繪圖迴圈 (The Factory)

這裡整合了 **Bar 圖 (差值)** 與 **Line 圖 (趨勢)** 的邏輯。

R

```
# ==============================================================================
# ==== B. 自動化繪圖迴圈 ====
# ==============================================================================

# 設定你的輸出路徑
save_dir <- "C:/Your/Output/Path/" 
if(!dir.exists(save_dir)) dir.create(save_dir, recursive = TRUE)

# 假設 measures_list 是你要跑的所有變數名稱
# 假設 plot_data_master 已經是整理好的 long format (含 Group, measure, Time/Trail, score)

for (m in unique(plot_data_master$measure)) {
  
  # 1. 篩選數據
  df_sum <- plot_data_master %>% 
    filter(measure == m) %>%
    group_by(Group, Time) %>% # Time 可能是 Pre/Post 或 ID
    summarise(
      avg = mean(score, na.rm = TRUE),
      se = sd(score, na.rm = TRUE) / sqrt(n()), 
      .groups = "drop"
    )

  if(nrow(df_sum) == 0) next
  
  # 2. 自動計算 Y 軸動態範圍 (關鍵步驟！)
  # ---------------------------------------------------------
  raw_max <- max(df_sum$avg + df_sum$se, na.rm = TRUE)
  raw_min <- min(df_sum$avg - df_sum$se, na.rm = TRUE)
  
  # 防呆：如果數據完全沒變異
  if(raw_max == raw_min) { raw_max <- raw_max + 0.1; raw_min <- raw_min - 0.1 }
  
  # 設定天花板：最高點 + 20% (留給星星的空間)
  y_limit_top <- raw_max + (raw_max - raw_min) * 0.2
  
  # 設定地板：如果是 Bar 圖，通常是 min 或 0；如果是 Line 圖，通常是 min - buffer
  # 這裡示範通用邏輯：往下留 5%
  y_limit_bottom <- raw_min - (raw_max - raw_min) * 0.05

  # 3. 判斷繪圖類型 (你可以根據 measure 名稱或外層迴圈切換)
  # ---------------------------------------------------------
  # 這裡示範：如果是兩點 (Pre/Post) 畫 Line；如果是差值 (Diff) 畫 Bar
  PLOT_TYPE <- "Line" # 或 "Bar"
  
  p <- ggplot(df_sum, aes(x = Time, y = avg, group = Group, fill = Group, color = Group))
  
  if (PLOT_TYPE == "Line") {
    # === [Line Plot 邏輯] ===
    p <- p +
      geom_line(linewidth = 1.2, position = position_dodge(width = 0.2)) + 
      geom_point(size = 4, position = position_dodge(width = 0.2), shape = 21, fill = "white", stroke = 2) + 
      geom_errorbar(aes(ymin = avg - se, ymax = avg + se), width = 0.1, linewidth = 0.8, position = position_dodge(width = 0.2)) +
      
      # X 軸：2點時往內擠 (5:4 視覺)
      scale_x_discrete(expand = expansion(mult = c(0.35, 0.35))) +
      
      # Y 軸：自動刻度 (scales::pretty_breaks 是神器)
      scale_y_continuous(
        breaks = scales::pretty_breaks(n = 5),
        limits = c(y_limit_bottom, y_limit_top),
        expand = expansion(mult = c(0, 0)) # 因為我們已經手算 limits，這裡設 0
      )
      
  } else if (PLOT_TYPE == "Bar") {
    # === [Bar Plot 邏輯] ===
    p <- p +
      geom_hline(yintercept = 0, color = "black", linewidth = 0.5) + # 0 線
      geom_bar(stat = "identity", width = 0.7, position = position_dodge(width = 0.8), color = "black", alpha = 0.9) +
      geom_errorbar(aes(ymin = avg - se, ymax = avg + se), width = 0.2, linewidth = 0.8, position = position_dodge(width = 0.8)) +
      
      # X 軸：Bar 圖通常左右留一點點就好
      scale_x_discrete(expand = expansion(mult = c(0.1, 0.1))) +
      
      # Y 軸：Bar 圖底部必須貼齊
      scale_y_continuous(
        breaks = scales::pretty_breaks(n = 5),
        # limits 不需設定，讓它自動適應，或者使用 coord_cartesian 裁切
        expand = expansion(mult = c(0, 0.15)) # 底部 0，頂部 15%
      )
  }

  # 4. 套用通用樣式
  # ---------------------------------------------------------
  p <- p +
    scale_fill_manual(values = my_colors) +
    scale_color_manual(values = my_colors) +
    labs(title = m, subtitle = "Mean ± SEM", x = NULL, y = "Score") +
    theme_scientific() # 呼叫我們定義好的主題

  # 5. 顯著性標記 (範例)
  # ---------------------------------------------------------
  # 假設你的統計結果叫 stat_res
  # gap <- calc_star_y(df_sum)
  # p <- p + geom_text(...) # 這裡放入你之前的標記邏輯
  
  # 6. 輸出 (純量控制)
  # ---------------------------------------------------------
  ggsave(
    filename = file.path(save_dir, paste0(m, "_", PLOT_TYPE, ".png")),
    plot = p,
    width = 4,  # 英寸 (4x5 = 5:4 比例)
    height = 5, 
    dpi = 300
  )
}
```

### 🔑 模板中的關鍵技術總結 (Cheat Sheet)

#### 1. 自動刻度生成器

不用再煩惱間距是 5 還是 10。

R

```
breaks = scales::pretty_breaks(n = 5)
```

#### 2. X 軸的空間美學

- **趨勢圖 (Line)**：要讓視覺集中，左右留白大。 `expand = expansion(mult = c(0.35, 0.35))`
    
- **柱狀圖 (Bar)**：要讓柱子飽滿，左右留白小。 `expand = expansion(mult = c(0.1, 0.1))`
    

#### 3. Y 軸的動態天花板

防止星星被切掉的唯一解法：**先算再畫**。

R

```
max_val <- max(avg + se)
y_limit <- max_val + (range * 0.2) # 留 20% 頭部空間
scale_y_continuous(limits = c(min, y_limit))
```

#### 4. 對齊的奧義

Bar 圖要對齊，這三個參數的 `width` 必須**完全一致**：

1. `geom_bar(position = position_dodge(width = 0.8))`
    
2. `geom_errorbar(..., position = position_dodge(width = 0.8))`
    
3. `geom_text(..., position = position_dodge(width = 0.8))`
